class ReservationPage extends StatefulWidget {
  const ReservationPage({super.key});

  @override
  State<ReservationPage> createState() => _ReserverState();
}

class _ReserverState extends State<ReservationPage> {
  String _villeDepart = "depart";
  String _villeArriver = "arriver";
  final TextEditingController _nomPassager = TextEditingController();

  @override
  void initState() {
    super.initState();
    context.read<TrajetBloc>().add(
      GetTrajetEvent(arriver: _villeArriver, depart: _villeDepart),
    );
  }

  void _ouvrirMenuVilles(
    BuildContext context,
    List<String> villesProvenantDuState,
    bool estDepart,
  ) async {
    // Si la liste est vide (pas encore charg√©e), on ne fait rien
    if (villesProvenantDuState.isEmpty) return;

    final String? choix = await showModalBottomSheet<String>(
      context: context,
      builder: (ctx) => ListView.builder(
        itemCount: villesProvenantDuState.length,
        itemBuilder: (context, index) => ListTile(
          title: Text(villesProvenantDuState[index]),
          onTap: () => Navigator.pop(ctx, villesProvenantDuState[index]),
        ),
      ),
    );

    if (choix != null) {
      setState(() {
        if (estDepart)
          _villeDepart = choix;
        else
          _villeArriver = choix;
      });

      // On relance la recherche du prix pour le nouveau trajet
      context.read<TrajetBloc>().add(
        GetTrajetEvent(depart: _villeDepart, arriver: _villeArriver),
      );
    }
  }

  int _selectedIndex = 0;
  DateTime? _dateSelector;
  Future<void> _ouvrirCalendrier() async {
    final DateTime? result = await showDatePicker(
      context: context,
      initialDate: DateTime.now(),
      firstDate: DateTime.now(),
      lastDate: DateTime.now().add(const Duration(days: 365)),
    );
    if (result != null) {
      setState(() {
        _dateSelector = result;
      });
    }
  }

  @override
  Widget build(BuildContext context) {
    List<String> listeVilles = [];
    return Scaffold(
      body: Container(
        // create: (context) => sl<TrajetBloc>(),
        child: BlocConsumer<TrajetBloc, Trajetstate>(
          listener: (context, state) {
            if (state is BilletLoaded) {
              ScaffoldMessenger.of(context).showSnackBar(
                SnackBar(
                  content: Text(
                    "Succes",
                    // style: TextStyle(color: Colors.green
                  ),
                  backgroundColor: Colors.green,
                ),
              );
              Navigator.pushReplacement(
                context,
                MaterialPageRoute(builder: (context) => PaiemenPage()),
              );
            }
            if (state is BilletFailure) {
              ScaffoldMessenger.of(context).showSnackBar(
                SnackBar(
                  content: Text(
                    "Echec",
                    // style: TextStyle(color: Colors.green
                  ),
                  backgroundColor: Colors.red,
                ),
              );
              Navigator.pushReplacement(
                context,
                MaterialPageRoute(builder: (context) => ReservationPage()),
              );
            }
          },
          builder: (context, state) {
            DateTime heure = DateTime.now();
            double montant = 0;
            if (state is TrajetLoaded) {
              heure = state.trajet.heureDepart;
              montant = state.trajet.prix;
            }
            return Container(
              child: Container(
                child: Stack(
                  children: [
                    Container(
                      decoration: BoxDecoration(
                        gradient: LinearGradient(
                          begin: Alignment.topCenter,
                          end: Alignment.bottomCenter,
                          colors: [Color(0xFFF25E7A), Color(0xFF44388C)],
                        ),
                      ),
                    ),
                    Positioned(
                      top: 40,
                      left: 20,
                      child: ProfileButton(
                        onTap: () => {showProfileMenu(context)},
                      ),
                    ),

                    SafeArea(
                      child: Column(
                        children: [
                          Container(),
                          Padding(
                            padding: const EdgeInsets.symmetric(vertical: 90),

                            child: Center(
                              child: Text(
                                "SAFARI",
                                style: GoogleFonts.monoton(
                                  fontSize: 40,
                                  fontWeight: FontWeight.bold,
                                  color: Colors.white,
                                  letterSpacing: 5,
                                ),
                              ),
                            ),
                          ),
                          Expanded(
                            child: Container(
                              width: double.infinity,
                              decoration: BoxDecoration(
                                color: Colors.white,
                                borderRadius: BorderRadius.only(
                                  topLeft: Radius.circular(40),
                                  topRight: Radius.circular(40),
                                ),
                              ),
                              child: SingleChildScrollView(
                                padding: EdgeInsets.all(30),
                                child: Column(
                                  children: [
                                    Text(
                                      " Reservation",
                                      style: TextStyle(
                                        fontSize: 24,
                                        fontWeight: FontWeight.bold,
                                      ),
                                    ),
                                    SizedBox(height: 30),
                                    //bunia
                                    RouteselectorWidget(
                                      departureCity: _villeDepart,
                                      arrivalCity: _villeArriver,
                                      oneDepartureTap: () => _ouvrirMenuVilles(
                                        context,
                                        listeVilles,
                                        true,
                                      ),
                                      oneArrrivalTap: () => _ouvrirMenuVilles(
                                        context,
                                        listeVilles,
                                        false,
                                      ),
                                    ),

                                    SizedBox(height: 5),
                                    Row(
                                      children: [
                                        Text("Depart"),
                                        SizedBox(width: 242),
                                        Text("Arrivee"),
                                      ],
                                    ),
                                    const SizedBox(height: 10),

                                    MyFieldForDate(
                                      onTap: _ouvrirCalendrier,
                                      selectorDate:
                                          _dateSelector ?? DateTime.now(),
                                      icon: Icons.calendar_view_day_rounded,
                                    ),
                                    SizedBox(height: 20),
                                    Column(
                                      children: [
                                        Row(
                                          children: [
                                            Text("Heure depart"),
                                            const SizedBox(width: 32),
                                            Text(heure.hour.toString()),
                                          ],
                                        ),
                                        const SizedBox(height: 10),
                                        Row(
                                          children: [
                                            Text("Montat a payer"),
                                            const SizedBox(width: 20),
                                            Text(montant.toDouble.toString()),
                                          ],
                                        ),
                                      ],
                                    ),
                                    const SizedBox(height: 20),
                                    MyTextFieldRectangle(
                                      hintText: "Nom du passager",
                                      icon: Icons.person,
                                      controller: _nomPassager,
                                    ),
                                    const SizedBox(height: 40),
                                    PrimaryButton(
                                      text: "RESERVEZ",
                                      onPressed: () {
                                        if (_nomPassager.text.isEmpty) {
                                          ScaffoldMessenger.of(
                                            context,
                                          ).showSnackBar(
                                            SnackBar(
                                              content: Text(
                                                "Le nom du passager est obligatoire",
                                              ),
                                            ),
                                          );
                                          return;
                                        }
                                        final state = context
                                            .read<TrajetBloc>()
                                            .state;
                                        if (state is TrajetLoaded) {
                                          context.read<BilletBloc>().add(
                                            acheterBilletSubmitted(
                                              trajetId: state.trajet.id,
                                              prix: state.trajet.prix,
                                            ),
                                          );
                                        }
                                      },
                                    ),
                                    const SizedBox(height: 75),
                                    CustomBottomNavBar(
                                      currentIndex: _selectedIndex,
                                      onTap: (index) {
                                        setState(() {
                                          _selectedIndex = index;
                                        });
                                      },
                                    ),
                                  ],
                                ),
                              ),
                            ),
                          ),
                        ],
                      ),
                    ),
                  ],
                ),
              ),
            );
          },
        ),
      ),
    );
  }
}
